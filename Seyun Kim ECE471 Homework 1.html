#Gaussian functional form used to model the noisy sinewave
#Function takes in x and gaussian parameters w, mu, sigma, and b in dictionary form
def fgaussian(x, params):
    phi = tf.exp(-(tf.transpose(x)-params["mu"])**2/params["sigma"]**2)
    return tf.transpose(tf.matmul(tf.transpose(params["w"]), tf.cast(phi, tf.float32)) + params["b"])

#initializing model parameters
params = dict()
params["w"] = tf.get_variable('w', [M,1], tf.float32, tf.random_normal_initializer())
params["mu"] = tf.get_variable('mu', [M,1], tf.float32, tf.random_uniform_initializer())
    # mu initialized from uniform distribution as x is drawn from uniform distribution
params["sigma"] = tf.get_variable('sigma', [M,1], tf.float32, tf.random_normal_initializer(), constraint = lambda x: tf.abs(x))
params["b"] = tf.get_variable('b', [], tf.float32, tf.zeros_initializer())

#Gaussian base function
def basis(x, mu, sigma):
    return np.exp(-(x-mu)**2/sigma**2)

#Loss(cost) function to be minimized
def loss(y, y_hat):
    return tf.reduce_mean(0.5*tf.pow(y-y_hat, 2))
